BÜTÜN RANDEVU BİLGİLERİ
DECLARE @a INT;
SELECT
    R.RandevuID,
    R.RandevuTarihi,
    R.RandevuSaati,
    CONCAT(D.DoktorAdi, ' ', D.DoktorSoyadi) AS DoktorAdSoyad,
    B.BolumAdi AS DoktorBolum
FROM
    Randevu R
INNER JOIN Doktor D ON R.DoktorID = D.DoktorID
INNER JOIN BolumAdi B ON D.DoktorBolumID = B.BolumID
WHERE
    R.HastaID = @a;
